create table board_days (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  board_id bigint not null REFERENCES public.boards(id) ON UPDATE CASCADE ON DELETE CASCADE,
  created_day int4 not null,
  UNIQUE (board_id, created_day),
  done bool not null,
  color_index int2 not null default 0,
  notes text not null default ''
);
